/** An active layer connector that delegates port registration and enumeration operations
 to a connector capsule instance. This class provides a bridge between the layer service
 infrastructure and a capsule-based connector implementation, allowing dynamic port
 registration and service discovery to be handled by an active capsule instance running
 in its own thread.
 @see RTLayerConnector is the base class for layer connectors.
 @see RTConnector_Actor is the capsule that handles the actual connector operations.
 */
class RTActiveConnector : public RTLayerConnector
{
public:
    /** Construct an active connector associated with a connector capsule.
     @param[in] connector The connector capsule instance that will handle registration
     and enumeration operations. Must not be NULL.
     */
    explicit              RTActiveConnector( RTConnector_Actor * );
    
    /** Destructor. Cleans up the active connector. */
    virtual               ~RTActiveConnector( void );

    /** Register a Service Access Point (SAP) with the layer service.
     This operation delegates to the associated connector capsule to perform the
     actual registration.
     @param[in] context The controller managing the capsule that owns the port.
     @param[in] port The protocol port to register as a SAP.
     @param[in] name The service name to register the SAP under.
     @return Layer data associated with the registered SAP, or NULL if registration fails.
     */
